function submitToGoogleForms(event, loanType) { event.preventDefault(); const googleFormUrl = "https://docs.google.com/forms/d/e/1FAIpQLSfcXwrNVqY0CiNY1hCJhRazXrIOUPi47lD0OUvypefOqTKuGw/formResponse"; let name, mobile, empStatus, amount, tenure; if(loanType === 'Home Loan') { name = document.getElementById('home-name').value; mobile = document.getElementById('home-mobile').value; empStatus = document.getElementById('home-emp').value; amount = document.getElementById('home-amount').value; tenure = document.getElementById('home-tenure').value; } else if(loanType === 'Business Loan') { name = document.getElementById('business-name').value; mobile = document.getElementById('business-mobile').value; empStatus = document.getElementById('business-emp').value; amount = document.getElementById('business-amount').value; tenure = document.getElementById('business-tenure').value; } else if(loanType === 'Personal Loan') { name = document.getElementById('personal-name').value; mobile = document.getElementById('personal-mobile').value; empStatus = document.getElementById('personal-emp').value; amount = document.getElementById('personal-amount').value; tenure = document.getElementById('personal-tenure').value; } else if(loanType === 'Loan Against Property') { name = document.getElementById('property-name').value; mobile = document.getElementById('property-mobile').value; empStatus = document.getElementById('property-emp').value; amount = document.getElementById('property-amount').value; tenure = document.getElementById('property-tenure').value; } // ब्राउज़र ब्लॉकिंग (CORS) से बचने के लिए Hidden Form सबमिशन const form = document.createElement('form'); form.method = 'POST'; form.action = googleFormUrl; form.target = 'hidden_iframe'; // आपके फॉर्म के 100% सटीक और असली एंट्री आईडी यहाँ डाल दिए हैं const data = { 'entry.1691230623': name, // Applicant Name 'entry.888796229': mobile, // Mobile Number 'entry.1018306060': loanType, // Type of Loan 'entry.685354921': amount, // Desired Loan Amount 'entry.1388656711': tenure, // Preferred Repayment Tenure 'entry.1223908869': empStatus // Current Employment Status }; for (const key in data) { if (data.hasOwnProperty(key)) { const hiddenField = document.createElement('input'); hiddenField.type = 'hidden'; hiddenField.name = key; hiddenField.value = data[key]; form.appendChild(hiddenField); } } // बैकग्राउंड में डेटा भेजने के लिए आईफ्रेम लॉजिक let iframe = document.getElementById('hidden_iframe'); if (!iframe) { iframe = document.createElement('iframe'); iframe.id = 'hidden_iframe'; iframe.name = 'hidden_iframe'; iframe.style.display = 'none'; document.body.appendChild(iframe); } document.body.appendChild(form); form.submit(); // सक्सेस मैसेज alert("धन्यवाद! आपकी " + loanType + " एप्लीकेशन सफलतापूर्वक दर्ज कर ली गई है। हमारी टीम जल्द ही आपसे संपर्क करेगी।"); event.target.reset(); document.body.removeChild(form); }
